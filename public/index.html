const pptxgen = require("pptxgenjs");
const path    = require("path");

async function buildSlides() {
  const pres  = new pptxgen();
  pres.layout = "LAYOUT_16x9";
  pres.title  = "Mi Presentación";

  // ── Diapositiva 1: Portada ──────────────────────────
  const s1 = pres.addSlide();
  s1.background = { color: "0D2B55" };

  s1.addText("Título Principal", {
    x: 1, y: 1.8, w: 8, h: 1.5,
    fontSize: 40, fontFace: "Arial Black",
    color: "FFFFFF", bold: true, align: "center",
  });

  s1.addText("Subtítulo de la presentación", {
    x: 1, y: 3.4, w: 8, h: 0.8,
    fontSize: 18, fontFace: "Calibri",
    color: "90CAF9", align: "center",
  });

  // ── Diapositiva 2: Contenido ────────────────────────
  const s2 = pres.addSlide();
  s2.background = { color: "F0F4F8" };

  s2.addShape(pres.shapes.RECTANGLE, {
    x: 0, y: 0, w: 10, h: 1.1,
    fill: { color: "1565C0" }, line: { type: "none" },
  });
  s2.addText("Slide de Contenido", {
    x: 0.4, y: 0.1, w: 9, h: 0.9,
    fontSize: 24, fontFace: "Arial Black",
    color: "FFFFFF", bold: true, valign: "middle",
  });
  s2.addText("• Punto 1\n• Punto 2\n• Punto 3", {
    x: 0.6, y: 1.4, w: 8.8, h: 3.5,
    fontSize: 18, fontFace: "Calibri", color: "1A237E",
  });

  // ── Guardar ─────────────────────────────────────────
  const outFile = process.env.PPTX_OUTPUT
    || path.join(__dirname, "presentacion.pptx");
  await pres.writeFile({ fileName: outFile });
  console.log("✅ Presentación creada: " + outFile);
}

buildSlides().catch(console.error);
